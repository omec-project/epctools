141  {
142  if (m_acquire)
143  m_mtx.enter();
144  }
146  ~EMutexLock()
147  {
148  if (m_acquire)
149  m_mtx.leave();
150  }
151 
155  Bool acquire(Bool wait = True)
156  {
157  if (!m_acquire)
158  m_acquire = m_mtx.enter(wait);
159  return m_acquire;
160  }
