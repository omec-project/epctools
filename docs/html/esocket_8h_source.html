773  Int recv()
774  {

776  // modified this routine to use a buffer allocated from the stack
777  // instead of a single buffer allocated from the heap (which had
778  // been used for both reading and writing) to avoid between the
779  // read and write process
780  //
781  UChar buf[2048];
782  Int totalReceived = 0;
783 
784  while (True)
785  {
786  Int amtReceived = ::recv(this->getHandle(), (PSNDRCVBUFFER)buf, sizeof(buf), 0);
787  if (amtReceived > 0)
788  {
789  m_rbuf.writeData(buf, 0, amtReceived);
790  totalReceived += amtReceived;
791  }
792  else if (amtReceived == 0)
793  {
794  setState( SocketState::Disconnected );

796  }
797  else
798  {
799  this->setError();
800  if (this->getError() == EWOULDBLOCK)
801  {

804  }
805  throw TcpTalkerError_UnableToRecvData();

812  Void send(Bool override = False)
813  {
814  UChar buf[2048];
815 
816  EMutexLock lck(m_sendmtx, False);
817  if (!lck.acquire(False))
818  return;
819 
820  if (!override && m_sending)
821  return;
822 
823  if (m_wbuf.isEmpty())
824  {
825  m_sending = false;
826  return;
827  }
828 
829  if (getState() != SocketState::Connected)
830  {
831  //std::raise(SIGINT);
832  throw TcpTalkerError_InvalidSendState(Base<TQueue,TMessage>::getStateDescription(getState()));
833  }
834 
835  m_sending = true;
836  while (true)
837  {
838  if (m_wbuf.isEmpty())
839  {
840  m_sending = false;
841  break;
842  }
843 
844  Int packetLength = 0;
845  Int amtRead = m_wbuf.peekData((pUChar)&packetLength, 0, sizeof(packetLength));
846  if (amtRead != sizeof(packetLength))
847  {
848  EString msg;
849  msg.format("expected %d bytes, read %d bytes", sizeof(packetLength), amtRead);
850  throw TcpTalkerError_ReadingWritePacketLength(msg.c_str());
851  }
852 
853  Int sentLength = 0;
854  while (sentLength < packetLength)
855  {
856  Int sendLength = packetLength - sentLength;
857  if (sendLength > (Int)sizeof(buf))
858  sendLength = sizeof(buf);
859 
860  // get data from the circular buffer
861  amtRead = m_wbuf.peekData((pUChar)buf, sizeof(packetLength) + sentLength, sendLength);
862  if (amtRead != sendLength)
863  {
864  EString msg;
865  msg.format("expected %d bytes, read %d bytes", sendLength, amtRead);
866  throw TcpTalkerError_ReadingWritePacketLength(msg.c_str());
867  }
868 
869  // write the data to the socket
870  Int amtWritten = send(buf, sendLength);
871  if (amtWritten == -1) // EWOULDBLOCK
872  break;
873 
874  sentLength += amtWritten;
875  if (amtWritten != sendLength) // only part of the data was written
876  break;
877  }
878 
879  packetLength -= sentLength;
880  m_wbuf.readData(NULL, 0, sentLength + (!packetLength ? sizeof(packetLength) : 0));
881  if (packetLength > 0)
882  {
883  // need to update the buffer indicating the amount of the
884  // message remaining in the circular buffer
885  //fprintf(stderr,"wrote %d bytes of %d\n", sentLength, packetLength + sentLength);
886  m_wbuf.modifyData((pUChar)&packetLength, 0, (Int)sizeof(packetLength));
887  break;
888  }
889  }
890  }
